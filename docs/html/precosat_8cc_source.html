164 template<class T, class L> class Sorter {
165  Mem & mem;
166  Stack<long> stack;
167  const static long limit = 10;
168  static void swap (T & a, T & b) { T tmp = a; a = b; b = tmp; }
169  static void cwap (T & a, T & b) {
170  if (L (b, a)) swap (a, b);
171  assert (!L (b, a)); // otherwise 'L' inconsistent
172  }
173  void part (T * a, long l, long r, long & i) {
174  long j = r; i = l - 1;
175  T pivot = a[j];
176  for (;;) {
177  while (L (a[++i], pivot)) ;
178  while (L (pivot, a[--j])) if (j == l) break;
179  if (i >= j) break;
180  swap (a[i], a[j]);
181  }
182  swap (a[i], a[r]);
183  }
184  void qsort (T * a, long n) {
185  long l = 0, r = n - 1;
186  if (r - l <= limit) return;
187  long m, ll, rr, i;
188  assert (!stack);
189  for (;;) {
190  m = (l + r) / 2;
191  swap (a[m], a[r-1]);
192  cwap (a[l], a[r-1]);
193  cwap (a[l], a[r]);
194  cwap (a[r-1], a[r]);
195  part (a, l+1, r-1, i);
196  if (i - l < r - i) ll = i+1, rr = r, r = i-1;
197  else ll = l, rr = i-1, l = i+1;
198  if (r - l > limit) {
199  assert (rr - ll > limit);
200  stack.push (mem, ll);
201  stack.push (mem, rr);
202  } else if (rr - ll > limit) l = ll, r = rr;
203  else if (stack) r = stack.pop (), l = stack.pop ();
204  else break;
205  }
206  }
207  void isort (T * a, long n) {
208  for (long i = n-1; i > 0; i--) cwap (a[i-1], a[i]);
209 #ifndef NDEBUG
210  for (long i = 1; i < n; i++) assert (!L (a[i], a[0]));
211 #endif
212  for (long i = 2; i < n; i++) {
213  long j = i;
214  T pivot = a[i];
215  while (L (pivot, a[j-1]))
216  a[j] = a[j-1], j--, assert (j > 0);
217  a[j] = pivot;
218  }
219  }
220  void check (T * a, long n) {
221 #ifndef NDEBUG
222  for (long i = 0; i < n-1; i++) assert (!L (a[i+1], a[i]));
223 #else
224  (void) a, (void) n;
225 #endif
226  }
227 public:
228  Sorter (Mem & m) : mem (m) { }
229  ~Sorter () { stack.release (mem); }
230  void sort (T * a, long n) {
231  if (!n) return;
232  qsort (a, n);
233  isort (a, n);
234  check (a, n);
235  }
236 };
